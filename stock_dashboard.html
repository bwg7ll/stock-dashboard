// ==================== Drag & Drop ====================

function setupDragDrop() {

const dropZone = document.getElementById('dropZone');

['dragenter', 'dragover', 'dragleave', 'drop'].forEach(eventName => {

dropZone.addEventListener(eventName, preventDefaults, false);

document.body.addEventListener(eventName, preventDefaults, false);

});

function preventDefaults(e) { e.preventDefault(); e.stopPropagation(); }

['dragenter', 'dragover'].forEach(eventName => {

dropZone.addEventListener(eventName, () => dropZone.classList.add('drag-over'), false);

});

['dragleave', 'drop'].forEach(eventName => {

dropZone.addEventListener(eventName, () => dropZone.classList.remove('drag-over'), false);

});

dropZone.addEventListener('drop', handleDrop, false);

}

 

function handleDrop(e) {

const dt = e.dataTransfer;

const files = dt.files;

handleFiles(files);

}

 

function handleFiles(files) {

for (let i = 0; i < files.length; i++) {

const file = files[i];

if (file.name.endsWith('.json')) {

const reader = new FileReader();

reader.onload = function(e) {

try {

const data = JSON.parse(e.target.result);

importJSONData(data);

} catch (err) { alert('JSON 解析失败: ' + err.message); }

};

reader.readAsText(file);

} else if (file.name.endsWith('.csv') || file.name.endsWith('.xlsx') || file.name.endsWith('.xls')) {

const reader = new FileReader();

reader.onload = function(e) { parseCSV(e.target.result); };

reader.readAsText(file);

}

}

}

function handleCSVUpload(input) {

const file = input.files[0];

if (!file) return;

const reader = new FileReader();

reader.onload = function(e) { parseCSV(e.target.result); };

reader.readAsText(file);

}

 

function parseCSV(text) {

const lines = text.split('\n').filter(l => l.trim());

if (lines.length < 2) return;

const delimiters = [',', '\t', ';'];

let delimiter = ',';

for (const d of delimiters) {

if (lines[0].split(d).length > 2) { delimiter = d; break; }

}

const headers = lines[0].split(delimiter).map(h => h.trim().replace(/"/g, '').toLowerCase());

const colMap = {

symbol: findColumn(headers, ['symbol', 'ticker', 'code', '股票代码']),

name: findColumn(headers, ['name', 'description', 'desc', '股票名称']),

quantity: findColumn(headers, ['quantity', 'shares', 'qty', 'position', '持仓数量']),

price: findColumn(headers, ['price', 'last price', 'current price', 'market price', '现价']),

cost: findColumn(headers, ['cost basis', 'avg cost', 'average cost', 'cost', '成本价']),

marketValue: findColumn(headers, ['market value', 'position value', 'value', '市值']),

dayChange: findColumn(headers, ['day change', 'daily p&l', 'daily pnl', '当日盈亏']),

unrealized: findColumn(headers, ['unrealized p&l', 'unrealized', '未实现盈亏'])

};

 

pendingImport = [];

const previewData = [];

for (let i = 1; i < lines.length; i++) {

const cols = lines[i].split(delimiter).map(c => c.trim().replace(/"/g, ''));

if (cols.length < 3) continue;

const symbol = cols[colMap.symbol] || '';

if (!symbol || symbol.toLowerCase().includes('total') || symbol.toLowerCase().includes('cash')) continue;

const quantity = parseFloat(cols[colMap.quantity]) || 0;

if (quantity === 0) continue;

let price = parseFloat(cols[colMap.price]) || 0;

let cost = parseFloat(cols[colMap.cost]) || 0;

let mv = parseFloat(cols[colMap.marketValue]) || 0;

let dayChg = parseFloat(cols[colMap.dayChange]) || 0;

if (price === 0 && mv > 0 && quantity > 0) price = mv / quantity;

if (cost === 0) cost = price;

const row = { symbol: symbol.toUpperCase(), name: cols[colMap.name] || symbol, quantity: quantity, price: price, cost: cost, dayChange: dayChg, accountId: document.getElementById('importAccountSelect').value };

pendingImport.push(row);

if (previewData.length < 5) previewData.push(row);

}

 

const previewDiv = document.getElementById('csvPreview');

const previewTable = document.getElementById('previewTable');

previewDiv.classList.remove('hidden');

previewTable.innerHTML = '<tr class="text-gray-400 border-b border-gray-700"><th class="px-3 py-2">代码</th><th class="px-3 py-2">数量</th><th class="px-3 py-2">现价</th><th class="px-3 py-2">成本</th></tr>' +

previewData.map(r => '<tr class="border-b border-gray-700/30"><td class="px-3 py-2">' + r.symbol + '</td><td class="px-3 py-2 text-right">' + r.quantity + '</td><td class="px-3 py-2 text-right">$' + r.price.toFixed(2) + '</td><td class="px-3 py-2 text-right">$' + r.cost.toFixed(2) + '</td></tr>').join('') +

(pendingImport.length > 5 ? '<tr><td colspan="4" class="px-3 py-2 text-center text-gray-500">...还有 ' + (pendingImport.length - 5) + ' 条</td></tr>' : '');

 

importMethod = 'csv';

document.getElementById('confirmImportBtn').textContent = '确认导入 ' + pendingImport.length + ' 条数据';

}

 

function findColumn(headers, candidates) {

for (const c of candidates) {

const idx = headers.findIndex(h => h.toLowerCase().includes(c.toLowerCase()));

if (idx !== -1) return idx;

}

return -1;

}

// ==================== Dashboard ====================

function updateDashboard() {

const filtered = getFilteredHoldings();

if (filtered.length === 0) {

document.getElementById('emptyState').classList.remove('hidden');

document.querySelector('table').classList.add('hidden');

} else {

document.getElementById('emptyState').classList.add('hidden');

document.querySelector('table').classList.remove('hidden');

}

 

let totalMV = 0, totalDayChange = 0, totalCost = 0, totalDayChangeBase = 0;

filtered.forEach(h => {

const mv = h.quantity * h.price;

const cost = h.quantity * h.cost;

const dayChg = h.dayChange !== undefined ? h.dayChange : 0;

totalMV += mv;

totalCost += cost;

totalDayChange += dayChg;

totalDayChangeBase += (mv - dayChg);

});

 

const totalUnrealized = totalMV - totalCost;

const totalUnrealizedPct = totalCost > 0 ? (totalUnrealized / totalCost) * 100 : 0;

const totalDayChangePct = totalDayChangeBase > 0 ? (totalDayChange / totalDayChangeBase) * 100 : 0;

 

document.getElementById('totalMarketValue').textContent = formatCurrency(totalMV);

const dayChangeEl = document.getElementById('totalDayChange');

dayChangeEl.textContent = (totalDayChange >= 0 ? '+' : '') + formatCurrency(totalDayChange);

dayChangeEl.className = 'text-3xl font-bold ' + (totalDayChange >= 0 ? 'positive' : 'negative');

document.getElementById('totalDayChangePercent').textContent = (totalDayChange >= 0 ? '+' : '') + totalDayChangePct.toFixed(2) + '%';

document.getElementById('totalDayChangePercent').className = 'text-sm mt-1 font-medium ' + (totalDayChange >= 0 ? 'positive' : 'negative');

document.getElementById('dayChangeBadge').textContent = totalDayChange >= 0 ? '↗ 上涨' : '↘ 下跌';

document.getElementById('dayChangeBadge').className = 'text-xs px-2 py-1 rounded ' + (totalDayChange >= 0 ? 'bg-emerald-500/20 text-emerald-400' : 'bg-red-500/20 text-red-400');

 

const unrealEl = document.getElementById('totalUnrealizedPnl');

unrealEl.textContent = (totalUnrealized >= 0 ? '+' : '') + formatCurrency(totalUnrealized);

unrealEl.className = 'text-3xl font-bold ' + (totalUnrealized >= 0 ? 'positive' : 'negative');

document.getElementById('totalUnrealizedPnlPercent').textContent = (totalUnrealized >= 0 ? '+' : '') + totalUnrealizedPct.toFixed(2) + '%';

document.getElementById('totalUnrealizedPnlPercent').className = 'text-sm mt-1 font-medium ' + (totalUnrealized >= 0 ? 'positive' : 'negative');

 

renderHoldingsTable(filtered);

renderCharts(filtered);

}

function renderHoldingsTable(data) {

const tbody = document.getElementById('holdingsTable');

const search = document.getElementById('searchInput').value.toUpperCase();

const sortMethod = document.getElementById('sortSelect').value;

let filtered = data.filter(h => h.symbol.toUpperCase().includes(search) || (h.name && h.name.toUpperCase().includes(search)));

filtered.sort((a, b) => {

if (sortMethod === 'value') return (b.quantity * b.price) - (a.quantity * a.price);

if (sortMethod === 'pnl') return ((b.quantity * (b.price - b.cost)) - (a.quantity * (a.price - a.cost)));

if (sortMethod === 'symbol') return a.symbol.localeCompare(b.symbol);

return 0;

});

 

tbody.innerHTML = filtered.map(h => {

const mv = h.quantity * h.price;

const cost = h.quantity * h.cost;

const unreal = mv - cost;

const unrealPct = h.cost > 0 ? ((h.price - h.cost) / h.cost) * 100 : 0;

const dayChg = h.dayChange !== undefined ? h.dayChange : 0;

const account = accounts.find(a => a.id === h.accountId);

const accIcon = account ? (account.broker === 'ibkr' ? '🏦' : account.broker === 'schwab' ? '🏛️' : account.broker === 'firstrade' ? '🌐' : '📁') : '📁';

return '<tr class="hover:bg-slate-800/50 transition">' +

'<td class="px-5 py-3 text-xs text-gray-400">' + accIcon + ' ' + (account ? account.name : h.accountId) + '</td>' +

'<td class="px-5 py-3 font-semibold">' + h.symbol + '</td>' +

'<td class="px-5 py-3 text-gray-400">' + (h.name || '-') + '</td>' +

'<td class="px-5 py-3 text-right">' + h.quantity.toLocaleString() + '</td>' +

'<td class="px-5 py-3 text-right">$' + h.price.toFixed(2) + '</td>' +

'<td class="px-5 py-3 text-right text-gray-400">$' + h.cost.toFixed(2) + '</td>' +

'<td class="px-5 py-3 text-right font-medium">$' + mv.toLocaleString('en-US', {minimumFractionDigits: 2, maximumFractionDigits: 2}) + '</td>' +

'<td class="px-5 py-3 text-right ' + (dayChg >= 0 ? 'positive' : 'negative') + '">' + (dayChg >= 0 ? '+' : '') + formatCurrency(dayChg) + '</td>' +

'<td class="px-5 py-3 text-right ' + (unreal >= 0 ? 'positive' : 'negative') + ' font-medium">' + (unreal >= 0 ? '+' : '') + formatCurrency(unreal) + '</td>' +

'<td class="px-5 py-3 text-right ' + (unrealPct >= 0 ? 'positive' : 'negative') + '">' + (unrealPct >= 0 ? '+' : '') + unrealPct.toFixed(2) + '%</td>' +

'<td class="px-5 py-3 text-center"><button onclick="openEdit(\'' + h.id + '\')" class="text-gray-500 hover:text-blue-400 transition text-sm">✏️</button></td>' +

'</tr>';

}).join('');

}

function renderCharts(data) {

const accountTotals = {};

data.forEach(h => {

const mv = h.quantity * h.price;

accountTotals[h.accountId] = (accountTotals[h.accountId] || 0) + mv;

});

const accountLabels = Object.keys(accountTotals).map(id => {

const acc = accounts.find(a => a.id === id);

return acc ? acc.name : id;

});

const accountData = Object.values(accountTotals);

const accountColors = ['#3b82f6', '#10b981', '#f59e0b', '#8b5cf6', '#ef4444'];

 

if (accountChart) accountChart.destroy();

accountChart = new Chart(document.getElementById('accountChart'), {

type: 'doughnut',

data: { labels: accountLabels, datasets: [{ data: accountData, backgroundColor: accountColors, borderWidth: 0 }] },

options: { responsive: true, maintainAspectRatio: false, plugins: { legend: { position: 'right', labels: { color: '#94a3b8', font: { size: 11 } } } } }

});

 

const sorted = [...data].sort((a, b) => (b.quantity * b.price) - (a.quantity * a.price)).slice(0, 10);

const holdingLabels = sorted.map(h => h.symbol);

const holdingData = sorted.map(h => h.quantity * h.price);

 

if (holdingChart) holdingChart.destroy();

holdingChart = new Chart(document.getElementById('holdingChart'), {

type: 'bar',

data: { labels: holdingLabels, datasets: [{ label: '市值 (USD)', data: holdingData, backgroundColor: '#3b82f6', borderRadius: 4 }] },

options: { responsive: true, maintainAspectRatio: false, plugins: { legend: { display: false } }, scales: { y: { ticks: { color: '#64748b', callback: v => '$' + (v/1000).toFixed(0) + 'k' }, grid: { color: 'rgba(255,255,255,0.05)' } }, x: { ticks: { color: '#94a3b8' }, grid: { display: false } } } }

});

}
